initialization:
	Install: anaconda
	-Go to anaconda prompt and type:
		conda create -n {virtual env}
		conda activate {virtual env}

	# installing poppler for pdf interaction and pdf2image for pdf to image conversion
	conda install -c conda-forge poppler
	pip install pdf2image

	# download tesseract-ocr from https://github.com/UB-Mannheim/tesseract/wiki
	# and add Tesseract OCR to the environment
	pip install pytesseract
	pip install pillow

	# Add the following to the code
	pytesseract.pytesseract.tesseract_cmd = 'C:\\Program Files\\Tesseract-OCR\\tesseract.exe'
